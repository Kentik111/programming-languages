// Функция подсчёта четных цифр чисел
func evenDigits(number: Int) -> Int {
    var count = 0 
    var num = number 
    
    while num > 0 { 
        let digit = num % 10 
        if digit % 2 == 0 { 
            count += 1 
        }
        num /= 10 
    }
    
    return count 
}

func main(){
    print("Enter number of numbers: ", terminator: "")
    // Ввод количества чисел и проверка на опечатки
    guard let n = Int(readLine() ?? "") else { 
            print("Do you know what a number is?") 
            return 
        }

    // Доп. условие - ограничение количества чисел
    if n > 10 {
            print("Too much numbers")
            return
        }

    // Количество должно быть больше нуля
    if n <= 0 {
        print("Please enter a positive number")
        return
    }
    
    // Ввод чисел и проверка на опечатки
    for _ in 0..<n {
        print("Enter number: ", terminator: "")
        guard let number = Int(readLine() ?? "") else { 
            print("Invalid input. Please try again") 
            return 
        }
    
        let evenDigitCount = evenDigits(number: number)  // вызов функции
        print("Even digits: \(evenDigitCount)") // вывод
    }
}
main()
