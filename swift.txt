import Foundation // подключение библиотеки для работы с Swift (содержит основные классы, функции и т.д. для работы с языком)

var N: Int = 0 // объявление переменной
print("Enter number: ", terminator: "") // Обращение к пользователю. terminator обозначает конец строки (в коде ввод числа не переходит на новую строку)
if let input = readLine(), let n = Int(input) { // ввод числа
    N = n
}
if N > 26 { // Проверяем, чтобы размер N не был больше количества букв в алфавите
    print("Error") // Выводим ошибку в противном случае
    exit(0) // Завершение программы
}

for i in 0..<N { // пробег шагов от 0 до заданного N
    var this_Char = UnicodeScalar("A").value + UInt32(N) - 1 // конечная буква
    for j in 0...i { // пробег индексов от 0 до i
        print(Character(UnicodeScalar(this_Char)!), terminator: " ") // Вывод букв с разделителем в виде пробела
        this_Char -= 1 // смена буквы (иначе будет выводиться одна и та же буква)
    }
    print() // переход на новую строку
}
