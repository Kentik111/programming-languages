import java.util.* // весь пакет (для работы с классами и т.д.)

// Функция проверки на панграмму
fun isPangram(sentence: String): Boolean {
    val letters = HashSet<Char>()
    
    for (str in sentence) {
        if (str.isLetter()) {
            letters.add(str.toLowerCase())
        }
    }
    
    return letters.size == 26
}

fun main() {
    val scanner = Scanner(System.`in`)
    print("Enter sentence: ")
    val sentence = scanner.nextLine() //ввод строки
    
    // символ должен быть только английской буквой или пробелом
    var englishSymbols = true
    for (symbol in sentence) {
        if (!symbol.isWhitespace() && !symbol.isLetter()) { 
            englishSymbols = false //иначе ложь
            break
        }
    }
    //проверка на отсутствие лишних символов
    if (!englishSymbols) {
        println("The sentence contains invalid characters")
        return
    }
    
    val result = isPangram(sentence) // вызов функции
    
    if (result) {
        println("true") // панграмма
    } else {
        println("false") // не панграмма
    }
}
