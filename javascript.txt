const readline = require('readline');
const rl = readline.createInterface({
    input: process.stdin, // ввод
    output: process.stdout // вывод
});

rl.question('Enter Number: ', (N) => { // обращение к пользователю и ввод числа
    N = parseInt(N); //преобразование числа N в целое
    if (N > 26) { // если размер N превышает количество букв в английском алфавите
        console.log("Error"); // Выводит ошибку
        rl.close(); // завершение программы
    } else { // если соответстует условию
        for (let i = 0; i < N; i++) { // пробег шагов от 0 до заданного N
            let this_Char = String.fromCharCode('A'.charCodeAt(0) + N - 1); // конечная буква
            for (let j = 0; j <= i; j++) { // пробег индексов от 0 до i
                process.stdout.write(this_Char + " "); // вывод буквы через пробел
                this_Char = String.fromCharCode(this_Char.charCodeAt(0) - 1); //смена буквы (иначе будет выводиться одна и та же буква)
            }
            console.log(); //переход на новую строку
        }
        rl.close(); //завершение программы
    }
