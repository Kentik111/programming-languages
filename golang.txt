package main // подключение пакета main

import "fmt" // подключение пакета ввода-вывода

func main() {
    var N int // инициализирование переменной (var - ключевое слово для объявления переменной)
 fmt.Print("Enter number: ")
 fmt.Scan(&N) // ввод числа
 if N > 26 { // если введено число, большее размера английского алфавита
     fmt.Println("Error") // ошибка
     return // завершение программы
 }

    for i := 0; i < N; i++ { // пробег шагов от 0 до заданного N
        this_Char := 'A' + N - 1 // конечная буква
        for j := 0; j <= i; j++ { // пробег букв от 0 до i
            fmt.Printf("%c ", this_Char) // вывод буквы
            this_Char-- // смена буквы (иначе будет выводиться одна и та же буква)
        }
        fmt.Println() // переход на новую строку
    }
}
