//функция подсчёта четных цифр
function evenDigits(number) { 
  let count = 0; // счётчик
  while (number > 0) { // цикл идёт, пока число не обнулится после делений
    let digit = number % 10; // получаем последнюю цифру в виде остатка после деления на 10
    if (digit % 2 === 0) { // если цифра делится на 2 без остатка
      count++; // счётчик увеличивается
    }
    number = Math.floor(number / 10); // делим  число без остатка на 10, чтобы продолжить цикл
  }
  return count; // возвращаем значение счетчика
}

number=evenDigits(number); //вызов функции
