import Control.Monad -- Подключение модуля для работы программы

main :: IO () -- Объявление функции main
main = do -- Начало блока do для выполнения последовательности действий ввода-вывода
  putStrLn "Enter number: " -- Обращение к пользователю
  n <- readLn :: IO Int -- Ввод числа
  if n > 26 -- Если размер n больше количества букв в алфавите
    then putStrLn "Error" -- Вывод ошибки
    else forM_ [0..n-1] $ \i -> do -- Цикл, который проходит от 0 до n-1, присваивая каждое значение переменной i (шаги)
      let thisChar = toEnum (fromEnum 'A' + n - 1 - i) :: Char -- Конечная буква
      forM_ [0..i] $ \j -> do -- Вложенный цикл, который проходит от 0 до i, присваивая каждое значение переменной j (буквы)
        putStr (toEnum (fromEnum 'A' + n - 1 - j) : " ") -- Вывод символа с разделителем в виде пробела
      putStrLn "" -- Переход на новую строку
