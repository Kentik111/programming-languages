use std::io; // Подключение стандартной библиотеки ввода-вывода

fn main() { // Начало функции main
    let mut input = String::new(); // Создание переменной (строка) для ввода пользователя
    println!("Enter number: "); // Обращение к пользователю
    io::stdin().read_line(&mut input).expect("Failed to read input"); // Читывается ввод пользователя с клавиатуры и сохраняется в переменной input. В случае ошибки при чтении выводится сообщение "Failed to read input".
    // Ввод, сохраненный в input, обрезается от возможных пробелов и преобразуется в тип данных u8 (беззнаковое целое 8 бит), ожидается, что пользователь ввел число, в противном случае будет выведено сообщение "Invalid input".
    let n: u8 = input.trim().parse().expect("Invalid input");

    if n > 26 { // Если размер N больше количества букв в алфавите
        println!("Error"); // Вывод ошибки
        return; // Завершение программы
    }

    for i in 0..n { // пробег шагов от 0 до заданного N
        let mut this_char = ('A' as u8 + n - 1) as char; // конечная буква
        for _ in 0..=i { // Пробег букв
            print!("{} ", this_char); // Вывод буквы с пробелом
            this_char = (this_char as u8 - 1) as char; // Смена буквы
        }
        println!(); // Переход на новую строку
    }
}
