# Функция для подсчёта количества чётных цифр в числе
def even_digits(number):
    count = 0
    while number > 0:
        digit = number % 10
        if digit % 2 == 0:
            count += 1
        number = number // 10
    return count

n = int(input("Enter a number of numbers: ")) # Ввод количества чисел

# Доп. условие - ограничение количества чисел
if n > 10:
    print("Too much numbers")
    exit()

for i in range(n):
    number = int(input("Enter a number: ")) # Ввод чисел

    number = even_digits(number) # Вызов функции
    print("Even digits: ", number) # Вывод
