print("Enter number") # Обращение к пользователю ввести число
n = int(input()) # Ввод числа
if n > 26: # Если N больше количества букв в английском алфавите
    print("Error") # Вывод ошибки
else: # Иначе
    for i in range(n): # Пробег шагов
        this_char = ord('A') + n - 1 # Конечная буква
        for j in range(i+1): # Пробег букв
            print(chr(this_char), end=" ") # Вывод букв через пробел
            this_char -= 1 # Смена буквы
        print() # Переход на новую строку
