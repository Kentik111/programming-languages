# Функция подсчёта четных цифр числа
def even_digits(number)
  count = 0 
  while number > 0
    digit = number % 10 
    if digit % 2 == 0 
      count += 1 
    end
    number = number / 10.floor 
  end
  return count 
end

puts "Enter a number of number:"
n = gets.chomp.to_i # ввод коичества чисел

# Проверка на опечатки
if n.zero?
  puts "Do you know what a number is?"
  exit
end

# Количество должно быть больше нуля
if n < 1
  puts "Invalid input. Please enter a positive number"
  return
end

# Доп. условие - ограничение количества чисел
if n > 10
  puts "Too much numbers"
  return
end

# ввод чисел
for i in 0...n
  puts "Enter number:" 
  number = gets.chomp.to_i
  if number.zero?
    puts "Invalid input. Please try again"
    exit
  end
  number = even_digits(number) # вызов функции
  puts "Even digits: #{number}" # вывод
end
